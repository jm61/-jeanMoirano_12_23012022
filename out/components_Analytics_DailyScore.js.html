/**
 * RadialBarChart User's score rendering
 * @param {number} score
 * @returns {JSX}
 */
 export const DailyScore = ({score}) => {
    const dailyValue = [
      {value: 1, fill: 'transparent'},
      {value: score, fill: '#FF0000'},
    ]
    return (
      <Container>
        <Heading>Score</Heading>
        <Caption>
          <Percentage>{100 * score}%</Percentage>
          <Text>
            de votre <br />
            objectif
          </Text>
        </Caption>
        <ResponsiveContainer width="90%" height={200}>
          <RadialBarChart
            startAngle={90}
            endAngle={500}
            innerRadius={40}
            outerRadius={100}
            barSize={10}
            data={dailyValue}>
            <RadialBar
              cornerRadius={50}
              dataKey="value"
              background={{fill: '#FCFCFC'}}
            />
          </RadialBarChart>
        </ResponsiveContainer>
      </Container>
    )
  }

// PropTypes
DailyScore.propTypes = {
  score: PropTypes.number
}
